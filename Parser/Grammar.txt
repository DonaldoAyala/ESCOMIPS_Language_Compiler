	;

function_type : 	VOID
	|		DATA_TYPE
	;

	;

code_block : 		'{' program '}'
	;

program : 		line ';'
	;

line : 			declaration
	|		statement
	|		assignment_expression
	;

declaration : 		DATA_TYPE ID
	|		DATA_TYPE ID '=' simple_expression
	;


assignment_expression : ID 

primary_expression : 	ID
	|		CONSTANT
	;

simple_expression : 	primary_expression OPERATOR primary_expression
	|		primary_expression
	;

expression : 		expression '(' expression ')'
	|		simple_expression
	;
